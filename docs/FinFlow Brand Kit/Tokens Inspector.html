10Como usar

Importa o variables.css no topo da árvore de estilos e usa var(--token) em qualquer lugar. Light/dark alterna no data-theme do <html>.

/* HTML root */
<html data-theme="light">   /* ou "dark" */

/* CSS — sempre via var() */
.my-card {
  background: var(--color-paper);
  border: 1px solid var(--color-border);
  border-radius: var(--radius-md);
  padding: var(--space-5);
  box-shadow: var(--shadow-card);
  color: var(--color-text-main);
}

/* JS — toggle entre modos */
const root = document.documentElement;
const theme = localStorage.getItem('finflow-theme') || 'light';
root.setAttribute('data-theme', theme);
